字符串中字符替换replace
    ${ProjectInfo}    Evaluate    u'测试项目一-二期项目;测试项目一-一期项目'

大小写转换
    ${projectGUID}    Set Variable    SDSDASDWWDQWD
    ${projectGUID}    Evaluate    '${projectGUID}'.lower()    #转换为小写sdsdasdwwdqwd
    log    ${projectGUID}
    ${projectGUID}    Evaluate    '${projectGUID}'.upper()    #转换为大写SDSDASDWWDQWD
    log    ${projectGUID}

空格处理strip
    # 去掉首尾空格
    ${String}    Set Variable    \ \ 项目 名称 = 项目A \ \
    log    ${String}    # 项目 名称 = 项目A
    ${String}    Evaluate    u'${String}'.strip()
    log    ${String}    #项目 名称 = 项目A
    # 去掉首空格
    ${String}    Set Variable    \ \ 项目 名称 = 项目A \ \
    log    ${String}
    ${String}    Evaluate    u'${String}'.rstrip()
    log    ${String}    #项目 名称 = 项目A \
    # 去掉尾空格
    ${String}    Set Variable    \ \ 项目 名称 = 项目A \ \
    log    ${String}
    ${String}    Evaluate    u'${String}'.lstrip()
    log    ${String}
    #去掉所有空格
    ${String}    Set Variable    \ \ 项目 名称 = 项目A \ \
    ${String}    Evaluate    u'${String}'.replace(' ','')
    log    ${String}    #项目名称=项目A
